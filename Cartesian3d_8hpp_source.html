325  inline int dsyevj3(double A[3][3], double Q[3][3], double w[3])
326  // ----------------------------------------------------------------------------
327  // Calculates the eigenvalues and normalized eigenvectors of a symmetric 3x3
328  // matrix A using the Jacobi algorithm.
329  // The upper triangular part of A is destroyed during the calculation,
330  // the diagonal elements are read but not destroyed, and the lower
331  // triangular elements are not referenced at all.
332  // ----------------------------------------------------------------------------
333  // Parameters:
334  // A: The symmetric input matrix
335  // Q: Storage buffer for eigenvectors
336  // w: Storage buffer for eigenvalues
337  // ----------------------------------------------------------------------------
338  // Return value:
339  // 0: Success
340  // -1: Error (no convergence)
341  // ----------------------------------------------------------------------------
342  {
343  const int n = 3;
344  double sd, so; // Sums of diagonal resp. off-diagonal elements
345  double s, c, t; // sin(phi), cos(phi), tan(phi) and temporary storage
346  double g, h, z, theta; // More temporary storage
347  double thresh;
348 
349  // Initialize Q to the identity matrix
350  for (int i = 0; i < n; i++) {
351  Q[i][i] = 1.0;
352  for (int j = 0; j < i; j++)
353  Q[i][j] = Q[j][i] = 0.0;
354  }
355 
356  // Initialize w to diag(A)
357  for (int i = 0; i < n; i++)
358  w[i] = A[i][i];
359 
360  // Calculate SQR(tr(A))
361  sd = 0.0;
362  for (int i = 0; i < n; i++)
363  sd += fabs(w[i]);
364  sd = sd * sd;
365 
366  // Main iteration loop
367  for (int nIter = 0; nIter < 50; nIter++) {
368  // Test for convergence
369  so = 0.0;
370  for (int p = 0; p < n; p++)
371  for (int q = p + 1; q < n; q++)
372  so += fabs(A[p][q]);
373  if (so == 0.0)
374  return 0;
375 
376  if (nIter < 4)
377  thresh = 0.2 * so / (n * n);
378  else
379  thresh = 0.0;
380 
381  // Do sweep
382  for (int p = 0; p < n; p++)
383  for (int q = p + 1; q < n; q++) {
384  g = 100.0 * fabs(A[p][q]);
385  if (nIter > 4 && fabs(w[p]) + g == fabs(w[p]) && fabs(w[q]) + g == fabs(w[q])) {
386  A[p][q] = 0.0;
387  }
388  else if (fabs(A[p][q]) > thresh) {
389  // Calculate Jacobi transformation
390  h = w[q] - w[p];
391  if (fabs(h) + g == fabs(h)) {
392  t = A[p][q] / h;
393  }
394  else {
395  theta = 0.5 * h / A[p][q];
396  if (theta < 0.0)
397  t = -1.0 / (sqrt(1.0 + theta * theta) - theta);
398  else
399  t = 1.0 / (sqrt(1.0 + theta * theta) + theta);
400  }
401  c = 1.0 / sqrt(1.0 + t * t);
402  s = t * c;
403  z = t * A[p][q];
404 
405  // Apply Jacobi transformation
406  A[p][q] = 0.0;
407  w[p] -= z;
408  w[q] += z;
409  for (int r = 0; r < p; r++) {
410  t = A[r][p];
411  A[r][p] = c * t - s * A[r][q];
412  A[r][q] = s * t + c * A[r][q];
413  }
414  for (int r = p + 1; r < q; r++) {
415  t = A[p][r];
416  A[p][r] = c * t - s * A[r][q];
417  A[r][q] = s * t + c * A[r][q];
418  }
419  for (int r = q + 1; r < n; r++) {
420  t = A[p][r];
421  A[p][r] = c * t - s * A[q][r];
422  A[q][r] = s * t + c * A[q][r];
423  }
424 
425  // Update eigenvectors
426  for (int r = 0; r < n; r++) {
427  t = Q[r][p];
428  Q[r][p] = c * t - s * Q[r][q];
429  Q[r][q] = s * t + c * Q[r][q];
430  }
431  }
432  }
433  }
434 
435  return -1;
436  }
